🔧 Google Sheets Setup Guide

Follow these steps once to connect your tracker to Google Sheets so applied jobs persist after refresh.

  1. Create a new Google Sheet. Name the first sheet Jobs.
  2. Add these column headers in Row 1:
    id | title | company | type | source | match | location | salary | posted | closureDate | status | applyUrl | desc | skills | notes | appliedAt
  3. Open Apps Script: In the Sheet, go to Extensions → Apps Script.
  4. Replace all code with the script below and click Save.
  5. Deploy as Web App: Click Deploy → New deployment → Web app. Set Execute as: Me and Who has access: Anyone. Copy the URL.
  6. Paste the URL in the "Sheet URL" field in the tracker and click Connect.
Apps Script Code (copy all)
const SHEET_NAME = "Jobs";

function doGet(e) {
  const action = e.parameter.action;
  if (action === "getAll") return getAll();
  if (action === "getStatuses") return getStatuses();
  return ContentService.createTextOutput(JSON.stringify({error:"unknown action"}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const data = JSON.parse(e.postData.contents);
  if (data.action === "updateStatus") return updateStatus(data);
  if (data.action === "upsertJob") return upsertJob(data);
  return ContentService.createTextOutput(JSON.stringify({error:"unknown action"}))
    .setMimeType(ContentService.MimeType.JSON);
}

function getAll() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(SHEET_NAME);
  const rows = sh.getDataRange().getValues();
  if (rows.length < 2) return json([]);
  const headers = rows[0];
  const data = rows.slice(1).map(r => {
    const obj = {};
    headers.forEach((h, i) => obj[h] = r[i]);
    return obj;
  });
  return json(data);
}

function getStatuses() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(SHEET_NAME);
  const rows = sh.getDataRange().getValues();
  if (rows.length < 2) return json({});
  const headers = rows[0];
  const idCol = headers.indexOf("id");
  const statusCol = headers.indexOf("status");
  const notesCol = headers.indexOf("notes");
  const closureCol = headers.indexOf("closureDate");
  const result = {};
  rows.slice(1).forEach(r => {
    result[String(r[idCol])] = {
      status: r[statusCol],
      notes: notesCol >= 0 ? r[notesCol] : "",
      closureDate: closureCol >= 0 ? r[closureCol] : ""
    };
  });
  return json(result);
}

function updateStatus(data) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(SHEET_NAME);
  const rows = sh.getDataRange().getValues();
  const headers = rows[0];
  const idCol = headers.indexOf("id");
  const statusCol = headers.indexOf("status");
  const notesCol = headers.indexOf("notes");
  const closureCol = headers.indexOf("closureDate");
  const appliedAtCol = headers.indexOf("appliedAt");
  for (let i = 1; i < rows.length; i++) {
    if (String(rows[i][idCol]) === String(data.id)) {
      if (statusCol >= 0) sh.getRange(i+1, statusCol+1).setValue(data.status);
      if (notesCol >= 0 && data.notes !== undefined) sh.getRange(i+1, notesCol+1).setValue(data.notes);
      if (closureCol >= 0 && data.closureDate !== undefined) sh.getRange(i+1, closureCol+1).setValue(data.closureDate);
      if (appliedAtCol >= 0 && data.status === "applied") sh.getRange(i+1, appliedAtCol+1).setValue(new Date().toISOString());
      return json({ok:true});
    }
  }
  // Row not found — append it
  return upsertJob({...data, action:"upsertJob"});
}

function upsertJob(data) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(SHEET_NAME);
  const headers = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0];
  const row = headers.map(h => data[h] !== undefined ? data[h] : "");
  sh.appendRow(row);
  return json({ok:true, appended:true});
}

function json(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
